    "## 4. Feature Engineering\n",
    "\n",
    "Creating new features that may better capture the physical reality of exoplanet detection.\n"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "id": "6a203632",
   "metadata": {},
   "outputs": [],
   "source": [
    "df_model = df.copy()\n",
    "\n",
    "# Feature 1: Transit Signal-to-Noise proxy\n",
    "# Deep transits of small stars are more significant\n",
    "df_model['snr_proxy'] = df_model['koi_depth'] / (df_model['koi_srad'] + 0.01)\n",
    "\n",
    "# Feature 2: Orbital compactness (period / stellar radius)\n",
    "# Very compact orbits around large stars \u2192 likely FP\n",
    "df_model['orbital_compactness'] = df_model['koi_period'] / (df_model['koi_srad'] + 0.01)\n",
    "\n",
    "# Feature 3: Planet-to-star radius ratio (scaled)\n",
    "df_model['radius_ratio'] = df_model['koi_prad'] / (df_model['koi_srad'] * 109.2 + 0.01)\n",
    "\n",
    "# Feature 4: Insolation class (ordinal)\n",
    "bins = [0, 0.25, 1.0, 4.0, 25.0, np.inf]\n",
    "labels = ['frozen','cold','habitable','warm','hot']\n",
    "df_model['insolation_class'] = pd.cut(df_model['koi_insol'],\n",
    "                                       bins=bins, labels=labels)\n",
    "\n",
